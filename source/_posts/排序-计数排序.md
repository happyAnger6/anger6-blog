def count_sort(a, start, end):  
    cnts = [0 for _ in range(start, end+1)]  
    for e in a:  
        cnts[e-start]+=1  
  
    print(cnts)  
    sum = 0  
    for i in range(1, len(a)):  
        cnts[i] += cnts[i-1]  
  
    new_a = a[:]  
    for e in a:  
        print(cnts, e-start)  
        new_a[cnts[e-start]-1] = e  
        cnts[e-start]-=1  
    return new_a

```python
def isHw(a):  
    i, j = 0, len(a)-1  
    while i <= j:  
        if a[i] != a[j]:  
            return False  
 i += 1  
        j -= 1  
  
    return True  
if __name__ == "__main__":  
    print(isHw('abcdedcba'))  
    print(isHw('abcdedcb'))  
    print(isHw('abcdeedcba'))
```
